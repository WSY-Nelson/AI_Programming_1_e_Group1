def isWordGuessed(secretWord, lettersGuessed):

    for c in set(secretWord):
        if not c in lettersGuessed:
            return False
    return True

def showWord(word, guessed):

    word = word.upper()
    word_hint = ["_" if ele not in guessed else ele for ele in word]
    return " ".join(word_hint)
